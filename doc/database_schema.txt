Person 
-----
id PK int
LOCAL_ID varchar(10)

AccessToken
-----
id PK int
ORCID_ID character(19) FK >- Person.ORCID_ID
SCOPE varchar(50)
ACCESS_TOKEN varchar(50)
TOKEN_TYPE varchar(10)
REFRESH_TOKEN varchar(50)
EXPIRES_IN bigint
JSON varchar(500)
CREATED timestamp

LogEntry

id PK int
